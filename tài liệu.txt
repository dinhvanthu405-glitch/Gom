        function kiemTraDuLieu() {
            let idsach = document.getElementById("idsach").value.trim();
            let tensach = document.getElementById("tensach").value.trim();
            let tacgia = document.getElementById("tacgia").value.trim();
            let gia = document.getElementById("gia").value;
            let uutien = document.getElementById("uutien").value;

            if (idsach === "" || tensach === "" || tacgia === "" || gia === "" || uutien === "") {
                alert("Không được để trống dữ liệu");
                return false;
            }

            if (idsach.length > 10) {
                alert("ID sách tối đa 10 ký tự");
                return false;
            }

            if (tensach.length > 30) {
                alert("Tên sách tối đa 30 ký tự");
                return false;
            }

            if (tacgia.length > 50) {
                alert("Tác giả tối đa 50 ký tự");
                return false;
            }

            if (uutien != 1 && uutien != 2) {
                alert("Nhóm ưu tiên chỉ được nhập 1 hoặc 2");
                return false;
            }

            let thanhTien = parseFloat(gia);
            if (uutien == 1) {
                thanhTien += gia * 0.02;
            } else {
                thanhTien += gia * 0.05;
            }

            document.getElementById("ketqua").innerHTML =
                "✅ Thêm sách thành công <br> Thành tiền: " + thanhTien + " VNĐ";

            return false; // không submit server
        }
    </script>
