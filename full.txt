import hashlib

# Hàm tìm nghịch đảo modulo
def mod_inverse(a, m):
    for i in range(1, m):
        if (a * i) % m == 1:
            return i
    return None

# Hàm băm
def hash_message(message):
    return int(hashlib.sha256(message.encode()).hexdigest(), 16)

# Nhập dữ liệu
p = int(input("Nhập p: "))
a = int(input("Nhập a: "))
x = int(input("Nhập khóa bí mật x: "))
k = int(input("Nhập số ngẫu nhiên k: "))
message = input("Nhập bản rõ: ")

# Tạo khóa công khai
y = pow(a, x, p)

# Băm thông điệp
H = hash_message(message)

# Tạo chữ ký
r = pow(a, k, p)

k_inv = mod_inverse(k, p - 1)

s = ((H - x * r) * k_inv) % (p - 1)

print("\n===== KHÓA =====")
print("Khóa công khai y =", y)
print("Khóa bí mật x =", x)

print("\n===== CHỮ KÝ =====")
print("r =", r)
print("s =", s)

# Xác thực
v1 = (pow(y, r, p) * pow(r, s, p)) % p
v2 = pow(a, H, p)

print("\n===== XÁC THỰC =====")
print("v1 =", v1)
print("v2 =", v2)

if v1 == v2:
    print("Chữ ký hợp lệ")
else:
    print("Chữ ký không hợp lệ")

Công thức ElGamal:

y=a^x \bmod p

r=a^k \bmod p

s=k^{-1}(H(m)-xr) \bmod (p-1)

a^{H(m)} \equiv y^r r^s \pmod p

# Hàm tìm khóa Vigenere

def tim_khoa_vigenere(plain, cipher):
    key = ""

    plain = plain.upper()
    cipher = cipher.upper()

    for p, c in zip(plain, cipher):
        if p.isalpha() and c.isalpha():
            k = (ord(c) - ord(p)) % 26
            key += chr(k + ord('A'))

    return key

def giai_ma_vigenere(cipher, key):
    result = ""
    key = key.upper()
    j = 0

    for char in cipher:
        if char.isalpha():
            k = ord(key[j % len(key)]) - ord('A')

            if char.isupper():
                p = (ord(char) - ord('A') - k) % 26
                result += chr(p + ord('A'))
            else:
                p = (ord(char) - ord('a') - k) % 26
                result += chr(p + ord('a'))

            j += 1
        else:
            result += char

    return result

def ma_hoa(text, k):
    result = ""

    for ch in text:
        if ch.isalpha():
            result += chr((ord(ch) - 65 + k) % 26 + 65)
        else:
            result += ch

    return result


def giai_ma(text, k):

16. Mã hóa và giải mã Vigenere



def vigenere_encrypt(text, key):
    result = ""
    key = key.upper()
    j = 0

    for ch in text.upper():
        if ch.isalpha():
            k = ord(key[j % len(key)]) - 65
            result += chr((ord(ch) - 65 + k) % 26 + 65)
            j += 1
        else:
            result += ch

    return result


def vigenere_decrypt(text, key):
    result = ""
    key = key.upper()
    j = 0

    for ch in text.upper():
        if ch.isalpha():
            k = ord(key[j % len(key)]) - 65
            result += chr((ord(ch) - 65 - k) % 26 + 65)
            j += 1
        else:
            result += ch

    return result


text = input("Nhập bản rõ: ")
key = input("Nhập khóa: ")

c = vigenere_encrypt(text, key)

print("Bản mã:", c)
print("Giải mã:", vigenere_decrypt(c, key))

20. Tạo và xác thực chữ ký ElGamal

k = int(input("k = "))
m = int(input("m = "))

y = pow(a, x, p)

r = pow(a, k, p)

for i in range(1, p):
    if (k * i) % (p - 1) == 1:
        k_inv = i
        break

s = ((m - x * r) * k_inv) % (p - 1)

print("Chữ ký:", (r, s))

v1 = (pow(y, r, p) * pow(r, s, p)) % p
v2 = pow(a, m, p)

if v1 == v2:
    print("Hợp lệ")
else:
    print("Không hợp lệ")
